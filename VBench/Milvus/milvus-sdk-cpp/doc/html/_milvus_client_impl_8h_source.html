174  template <typename Request, typename Response>
175  Status
176  apiHandler(std::function<Status(void)> validate, std::function<Request(void)> pre,
177  Status (MilvusConnection::*rpc)(const Request&, Response&),
178  std::function<Status(const Response&)> wait_for_status, std::function<void(const Response&)> post) {
179  if (connection_ == nullptr) {
180  return {StatusCode::NOT_CONNECTED, "Connection is not ready!"};
181  }
182 
183  if (validate) {
184  auto status = validate();
185  if (!status.IsOk()) {
186  return status;
187  }
188  }
189 
190  Request rpc_request = pre();
191  Response rpc_response;
192  auto status =
193  std::bind(rpc, connection_.get(), std::placeholders::_1, std::placeholders::_2)(rpc_request, rpc_response);
194  if (!status.IsOk()) {
195  // resp's status already checked in connection class
196  return status;
197  }
198 
199  if (wait_for_status) {
200  status = wait_for_status(rpc_response);
201  }
202 
203  if (status.IsOk() && post) {
204  post(rpc_response);
205  }
206  return status;
207  }
