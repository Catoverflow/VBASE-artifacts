83 template <typename T, DataType Dt>
84 typename std::enable_if<DataTypeTraits<Dt>::is_vector, StatusCode>::type
85 AddElement(const T& element, std::vector<T>& array) {
86  if (element.empty()) {
87  return StatusCode::VECTOR_IS_EMPTY;
88  }
89 
90  if (!array.empty() && element.size() != array.at(0).size()) {
91  return StatusCode::DIMENSION_NOT_EQUAL;
92  }
93 
94  array.emplace_back(element);
95  return StatusCode::OK;
96 }
