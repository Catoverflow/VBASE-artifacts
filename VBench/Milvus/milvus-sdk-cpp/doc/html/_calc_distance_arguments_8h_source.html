252  Status
253  Validate() const {
254  if (nullptr == vectors_left_ || vectors_left_->Count() == 0) {
255  return {StatusCode::INVALID_AGUMENT, "Vectors on the left of operator cannot be empty!"};
256  }
257 
258  if (nullptr == vectors_right_ || vectors_right_->Count() == 0) {
259  return {StatusCode::INVALID_AGUMENT, "Vectors on the right of operator cannot be empty!"};
260  }
261 
262  // To calculate distance, vector type must be equal.
263  // If user specified id array, the CalcDistance API will get collection schema to verify.
264  if (IsVectorType(vectors_left_->Type()) && IsVectorType(vectors_right_->Type())) {
265  if (vectors_left_->Type() != vectors_right_->Type()) {
266  return {StatusCode::INVALID_AGUMENT, "Vector types of left and right do not equal!"};
267  }
268 
269  static const std::set<std::string> float_types = {"L2", "IP"};
270  if (vectors_left_->Type() == DataType::FLOAT_VECTOR && float_types.find(metric_) == float_types.end()) {
271  return {StatusCode::INVALID_AGUMENT, "Invalid metric type for float vectors!"};
272  }
273 
274  static const std::set<std::string> binary_types = {"HAMMING", "TANIMOTO"};
275  if (vectors_left_->Type() == DataType::BINARY_VECTOR && binary_types.find(metric_) == binary_types.end()) {
276  return {StatusCode::INVALID_AGUMENT, "Invalid metric type for binary vectors!"};
277  }
278  }
279 
280  return Status::OK();
281  }
